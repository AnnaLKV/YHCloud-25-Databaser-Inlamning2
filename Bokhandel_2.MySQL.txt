-- Skapa trigger som minskar lagersaldo efter en order
DELIMITER $$

CREATE TRIGGER lageruppdatering
AFTER INSERT ON Orderrader
FOR EACH ROW
BEGIN
	UPDATE Produkter 
    SET Lagerstatus = Lagerstatus - NEW.Antal
    WHERE ISBN = NEW.ISBN;
END $$

DELIMITER ;


-- Skapa tabell Kundlogg (inför att skapa trigger som loggar när ny kund registreras)
CREATE TABLE Kundlogg (
id INT AUTO_INCREMENT PRIMARY KEY,
    KundID INT,
    Handelse VARCHAR(100),
    Skapad TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


-- Skapa trigger som loggar när ny kund registreras
DELIMITER $$

CREATE TRIGGER registrering_ny_kund
AFTER INSERT ON Kunder
FOR EACH ROW
BEGIN
    INSERT Kundlogg (KundID)
    VALUES (NEW.KundID);  
END $$

DELIMITER ;
